CREATE STOGROUP P3DSG VOLUMES(XPND01) VCAT P3DCAT;
COMMIT;
CREATE DATABASE PRICE3D STOGROUP P3DSG;
COMMIT;
CREATE TABLESPACE TSFIL
    IN PRICE3D
    USING STOGROUP P3DSG
    PRIQTY 10000
    SECQTY 1000
    SEGSIZE 32
    BUFFERPOOL BP0;
COMMIT;
CREATE TABLE FILMNT
    (
       ITEMID VARCHAR(20) NOT NULL,
       ITTYPE VARCHAR(10),
       ITCOLOR VARCHAR(10),
       ITWEIGH INT,
       ITUSED INT,
       ITREMAIN INT,
       ITPURCH REAL,
       ITVENDOR VARCHAR(30),
       ITVENDORID VARCHAR(20),
       ITVREORDER VARCHAR(70),
       ITLASTDT DATE,
       CONSTRAINT PK_FILMNT PRIMARY KEY (ITEMID)
    )
    IN PRICE3D.TSFIL;
COMMIT;
CREATE UNIQUE INDEX SCOTT.XFILMNT_PK
   ON SCOTT.FILMNT
     (ITEMID ASC)
     USING STOGROUP P3DSG
     PRIQTY 512
     SECQTY 64
     ERASE YES
BUFFERPOOL BP0;
COMMIT;
